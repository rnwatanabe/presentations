\section{Processamento de sinais}
%-------------------------------------------%
\begin{frame} 
\frametitle{\textbf{O uso de processamento de sinais}}
	 \begin{itemize}
	  \onslide<1->\item  No estudo do movimento humano podem ser medidos diferentes sinais: EEG, EMG, força, torque, ângulos.
	  
	  \onslide<2->\item Com isso, surgem diferentes questões de como esses sinais se relacionam:
	  
        \begin{itemize}
            \onslide<3->\item Existe alguma relação entre esses sinais?
            
            \onslide<4->\item Esta relação é linear ou não-linear?
            
            \onslide<5->\item Qual a relação de causalidade entre um sinal e outro?
        \end{itemize}
     
     \onslide<6->\item Estas perguntas não são respondidas de forma intuitiva ou com técnicas simples de análise de sinais.
	 \end{itemize}
\end{frame}
